async function generate() { const prompt = document.getElementById('user-prompt').value; const resultDiv = document.getElementById('result'); const apiKey = "ضع_مفتاح_الـ_API_الخاص_بك_هنا"; // تنبيه: لا تشارك هذا المفتاح مع أحد! if (!prompt) { alert("اكتب وش تبي الذكاء الاصطناعي يسوي!"); return; } resultDiv.innerText = "جاري التفكير وتوليد النص... 🧠"; try { // إرسال الطلب إلى OpenAI (محرك GPT) const response = await fetch('https://api.openai.com/v1/chat/completions', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${apiKey}` }, body: JSON.stringify({ model: "gpt-3.5-turbo", // أو gpt-4 إذا حسابك يدعمه messages: [{ role: "user", content: prompt }], temperature: 0.7 }) }); const data = await response.json(); // عرض النتيجة اللي رجعت من الذكاء الاصطناعي const aiResponse = data.choices[0].message.content; resultDiv.innerText = aiResponse; } catch (error) { console.error("خطأ:", error); resultDiv.innerText = "للاسف حدث خطأ في الاتصال. تأكد من مفتاح الـ API."; } }
